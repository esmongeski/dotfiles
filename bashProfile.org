  # Cache for gh PR hyperlink.
  # gh pr view runs in the background so it never blocks the prompt.
  # Results are stored in $HOME/.cache/gh_pr/ and survive across shell sessions.
  _gh_available=""
  _gh_pr_cache_branch=""
  _gh_pr_result=""

  # Read the current git branch by walking up to .git — no git process fork.
  # Handles both regular repos (.git dir) and worktrees (.git file with gitdir pointer).
  function _git_head_branch() {
      local _d="$PWD" _head="" _git_dir=""
      while [[ "$_d" != "" ]]; do
          if [[ -f "$_d/.git/HEAD" ]]; then
              _git_dir="$_d/.git"
              break
          elif [[ -f "$_d/.git" ]]; then
              local _ptr
              IFS= read -r _ptr < "$_d/.git"
              _git_dir="${_ptr#gitdir: }"
              # Resolve relative path
              [[ "$_git_dir" != /* ]] && _git_dir="$_d/$_git_dir"
              break
          fi
          [[ "$_d" == "/" ]] && return
          _d="${_d%/*}"
      done
      [[ -z "$_git_dir" ]] && return
      IFS= read -r _head < "$_git_dir/HEAD" 2>/dev/null
      [[ "$_head" == "ref: refs/heads/"* ]] && printf '%s' "${_head#ref: refs/heads/}"
  }

  function _get_gh_pr_url() {
      _gh_pr_result=""
      if [ -z "$_gh_available" ]; then
          command -v gh &>/dev/null && _gh_available="yes" || _gh_available="no"
      fi
      [ "$_gh_available" = "no" ] && return

      local _branch
      _branch=$(_git_head_branch)
      [ -z "$_branch" ] || [ "$_branch" = "main" ] || [ "$_branch" = "master" ] && return

      local _cache_file="$_GH_PR_CACHE_DIR/${_branch//\//_}"

      if [ "$_branch" != "$_gh_pr_cache_branch" ]; then
          _gh_pr_cache_branch="$_branch"
          # Fire-and-forget: fetch in background; only write the cache if we got a result.
          mkdir -p "$_GH_PR_CACHE_DIR"
          ( _u=$(gh pr view --json url -q .url 2>/dev/null); [ -n "$_u" ] && printf '%s' "$_u" > "$_cache_file" ) &
          disown
      fi

      # Read with builtin instead of forking cat.
      [[ -f "$_cache_file" ]] && IFS= read -r _gh_pr_result < "$_cache_file"
  }

  function _update_ps1() {
      local _exit=$?

      local _pr_url="$_gh_pr_result"
      local _modules="cwd,git,exit"

          export POWERLINE_GH_PR="🔗"
          _modules="cwd,shell-var,git,exit"
      else
          export POWERLINE_GH_PR=""
      fi
      # Count jobs without forking wc: remove newlines, compare lengths.
      local _jp _jc=0
      _jp=$(jobs -p)
      [[ -n "$_jp" ]] && { local _stripped="${_jp//$'\n'/}"; _jc=$(( ${#_jp} - ${#_stripped} + 1 )); }
      local _pl_out
      _pl_out="$($GOPATH/bin/powerline-go -error $_exit -jobs $_jc -shell bash -modules "$_modules" -shell-var POWERLINE_GH_PR -cwd-max-dir-size 3 -truncate-segment-width 5 -path-aliases=$EXTRA_POWERLINE_DIR_ALIASES  $EXTRA_POWERLINE_OPTIONS | sed -E 's|es/||; s/([a-zA-Z0-9_./-]{10})[a-zA-Z0-9_./-]+/\1…/')"
      if [ -n "$_pr_url" ]; then
          local _link=$'\e]8;;'"${_pr_url}"$'\a🔗\e]8;;\a'
          PS1="${_pl_out/🔗/$_link}"
      else
          PS1="$_pl_out"
      fi
      # Uncomment the following line to automatically clear errors after showing
      # them once. This not only clears the error for powerline-go, but also for
      # everything else you run in that shell. Don't enable this if you're not
      # sure this is what you want.

      set "?"
  }

** gchlike
This thing is super great. Give it a short string, and it greps your git branches for that string. If it finds exactly
one match, it checks out that branch. Otherwise it tells you what it did find you can refine your query (or just copy
the branch you want)
#+begin_src bash
function gchlike {
    local MATCHES=`g branch | grep -i $1 | awk '{print $1;}'`
    local MATCHCOUNT=`echo "$MATCHES" | wc -l`
    if [ "$MATCHCOUNT" -gt 1 ]; then
        echo "Too many branches!"
        echo "$MATCHES"
    elif [ "$MATCHES" == "" ]; then
        echo "No branches!"
    else
        echo "checking out $MATCHES"
        g checkout "$MATCHES"
    fi
}
#+end_src

*** Main, but not always main
Ok so, to get a little political, sometimes there are movements to change the language we use. And sometimes, people don't like it because changing language introduces cognitive tax, which is like, kinda understandable.
And to get more political, I think it's important to empathize with people that feel that way, even if you would prefer they change their language (which believe me, I frequently do, and in moments of impatience, wish people would just think a little harder).
I do believe language shapes how we think, and changing it can change how we think, and that's important.

This is a bit rambly, but tl;dr this all kinda manifests in this next function. I appreciate the move from master->main in git lexicon. But working in an environment that is inconsistent on which
represents the "branch with the closest-to-production-code" is a frequent, albeit minor, inconvenience (i.e. a cognitive tax). So I made this function to figure it out for me.
#+begin_src bash
  function masterOrMain {
    RESULT=`git rev-parse --verify main`
    if [ -z $RESULT ];
    then
            echo "master"
    else
            echo "main"
    fi
  }

* partyify
All credit to Sean Ezrol for this. Script that takes an image/gif and makes it have the party colors.
#+begin_src bash
function partyify {
    while [[ $# -gt 1 ]]
    do
    key="$1"
      case $key in
      -i|--input-file)
      INPUTFILE=$2
      shift
      ;;
      -c|-color)
      COLOR=$2
      shift
      ;;
      -f|--fuzz)
      FUZZ=$2
      shift
      ;;
      -o|--output-file)
      OUTPUTFILE=$2
      shift
      ;;
      *)
        # unknown arg
      ;;
    esac
    shift
    done

    echo Input - "${INPUTFILE}"
    party_colors=("#93FE90" "#8FB3FC" "#CF7CFA" "#EF4CEF" "#F1586A" "#F9D48D")
    for i in "${!party_colors[@]}"
    do
      echo   magick convert "${INPUTFILE}" -fill "${party_colors[i]}" -fuzz "${FUZZ}"% -opaque "${COLOR}" party_temp-"$((i+1))".png
      magick convert "${INPUTFILE}" -fill "${party_colors[i]}" -fuzz "${FUZZ}"% -opaque "${COLOR}" party_temp-"$((i+1))".png
      echo Making party_temp-"$((i+1))".png, replacing "${COLOR}" with "${party_colors[i]}"
    done
    magick convert party_temp-%d.png[1-"${#party_colors[@]}"] -set delay 10 -loop 0 "${OUTPUTFILE}"
    echo "${OUTPUTFILE} has been created."
}
